Bubble sort in golang

        func bubbleSort(arr []int) {
          n := len(arr)
          for i := 0; i < n-1; i++ {
            // Last i elements are already in place
            for j := 0; j < n-i-1; j++ {
              // Swap if the current element is greater than the next element
              if arr[j] > arr[j+1] {
                arr[j], arr[j+1] = arr[j+1], arr[j]
              }
            }
          }
        }
